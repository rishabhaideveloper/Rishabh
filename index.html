#!/bin/bash

apply_theme() {
  cat > ~/.termux/colors.properties << EOL
background=$1
foreground=$2
color0=$3
color1=$4
color2=$5
color3=$6
color4=$7
color5=$8
color6=$9
color7=${10}
color8=${11}
color9=${12}
color10=${13}
color11=${14}
color12=${15}
color13=${16}
color14=${17}
color15=${18}
EOL
  termux-reload-settings
  echo -e "\n\033[1;32m[$19 Theme Applied Successfully]\033[0m"
}

echo -e "\nSelect a Theme:"
echo "1) Pink"
echo "2) Blue"
echo "3) Green"
echo "4) Red"
echo "5) Dark"
read -p "Enter choice [1-5]: " ch
